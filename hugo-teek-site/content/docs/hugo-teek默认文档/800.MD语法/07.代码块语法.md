```python
from typing import List, Optional
from dataclasses import dataclass

@dataclass
class Product:
    id: int
    name: str
    price: float
    category: str

class ProductRepository:
    def __init__(self):
        self.products: List[Product] = []
    
    def add(self, product: Product) -> None:
        self.products.append(product)
    
    def find_by_category(self, category: str) -> List[Product]:
        return [p for p in self.products if p.category == category]
    
    def find_by_id(self, product_id: int) -> Optional[Product]:
        for product in self.products:
            if product.id == product_id:
                return product
        return None
```

#### Rust
```rust
use std::collections::HashMap;

#[derive(Debug, Clone)]
pub struct User {
    id: u64,
    username: String,
    email: String,
    active: bool,
}

pub struct UserRepository {
    users: HashMap<u64, User>,
    next_id: u64,
}

impl UserRepository {
    pub fn new() -> Self {
        Self {
            users: HashMap::new(),
            next_id: 1,
        }
    }

    pub fn create(&mut self, username: String, email: String) -> &User {
        let id = self.next_id;
        self.next_id += 1;
        
        let user = User {
            id,
            username,
            email,
            active: true,
        };
        
        self.users.insert(id, user);
        self.users.get(&id).unwrap()
    }

    pub fn find_by_id(&self, id: u64) -> Option<&User> {
        self.users.get(&id)
    }

    pub fn find_active(&self) -> Vec<&User> {
        self.users
            .values()
            .filter(|u| u.active)
            .collect()
    }
}

fn main() {
    let mut repo = UserRepository::new();
    
    let user = repo.create(
        String::from("张三"),
        String::from("zhangsan@example.com")
    );
    
    println!("创建用户: {:?}", user);
}
```

#### Kotlin
```kotlin
package com.example.app

import kotlinx.coroutines.*
import java.time.LocalDateTime

data class User(
    val id: Long,
    val username: String,
    val email: String,
    val createdAt: LocalDateTime = LocalDateTime.now()
)

class UserRepository {
    private val users = mutableListOf<User>()
    private var nextId = 1L

    fun create(username: String, email: String): User {
        require(username.isNotBlank()) { "用户名不能为空" }
        require(email.contains("@")) { "邮箱格式不正确" }

        val user = User(
            id = nextId++,
            username = username,
            email = email
        )
        users.add(user)
        return user
    }

    fun findById(id: Long): User? = users.find { it.id == id }

    fun findAll(): List<User> = users.toList()

    suspend fun fetchUserAsync(id: Long): User? = withContext(Dispatchers.IO) {
        delay(100) // 模拟网络延迟
        findById(id)
    }
}

// 使用示例
suspend fun main() {
    val repository = UserRepository()
    
    val user = repository.create("张三", "zhangsan@example.com")
    println("创建用户: $user")
    
    val fetchedUser = repository.fetchUserAsync(user.id)
    println("获取用户: $fetchedUser")
}
```
